### 点击某个元素 使页面滚动到指定位置


    ```
    Element(id='top') //需要显示的元素

    footer(@click="scrollView") //按钮
    
    methods:{
        scrollView(){
            //滚动方法
             document.getElementById('top').scrollIntoView({
                block: 'start',
                inline: 'nearest',
                behavior: 'smooth',
            })
        }
    }

    ```
