/**

* Definition for singly-linked list.
* function ListNode(val, next) {
*     this.val = (val===undefined ? 0 : val)
*     this.next = (next===undefined ? null : next)
* }
 */
/**
* @param {ListNode} head
* @return {boolean}
 */
var isPalindrome = function(head) {
    const valArr = []

//便利一次放到数组中 
    while(head){
        valArr.push(head.val)
        head = head.next
    }

    //双指针判断
    let left = 0
    let right = valArr.length - 1

    while(left <= right){
        if(valArr[left] === valArr[right]){
            left++
            right--
        }else{
            return false
        }
    }
    return true
};
