公司用一个字符串来表示员工的出勤信息

absent：缺勤
late：迟到
leaveearly：早退
present：正常上班
现需根据员工出勤信息，判断本次是否能获得出勤奖，能获得出勤奖的条件如下：

缺勤不超过一次；
没有连续的迟到/早退；
任意连续7次考勤，缺勤/迟到/早退不超过3次。
输入描述
用户的考勤数据字符串

记录条数 >= 1；
输入字符串长度 < 10000；
不存在非法输入；
如：

2
present
present absent present present leaveearly present absent

输出描述
根据考勤数据字符串，如果能得到考勤奖，输出”true”；否则输出”false”，
对于输入示例的结果应为：

true false

```javascript
// 引入node的输入模块，处理多行输入
const readline = require('readline').createInterface({
    input: process.stdin,
    output: process.stdout
});

/**
 * 辅助函数1：判断当前考勤项是否是【违规考勤】(迟到/早退)
 * @param {string} item 单个考勤项，如：late/leaveearly/present/absent
 * @returns {boolean} 是违规考勤返回true，否则false
 */
function isViolation(item) {
    return item === 'late' || item === 'leaveearly';
}

/**
 * 辅助函数2：判断当前考勤项是否是【不良考勤】(缺勤/迟到/早退)
 * @param {string} item 单个考勤项
 * @returns {boolean} 是不良考勤返回true，否则false
 */
function isBadRecord(item) {
    return item === 'absent' || isViolation(item);
}

/**
 * 核心校验函数：传入单条考勤的数组，判断是否能拿出勤奖
 * @param {string[]} recordArr 考勤数组 如：['present','absent','late']
 * @returns {boolean} 能拿奖返回true，否则false
 */
function checkAward(recordArr) {
    const len = recordArr.length;
    // ---------------------- 条件1：缺勤absent不超过1次 ----------------------
    const absentCount = recordArr.filter(item => item === 'absent').length;
    if (absentCount > 1) {
        return false;
    }

    // ---------------------- 条件2：无连续的迟到/早退 ----------------------
    for (let i = 1; i < len; i++) {
        // 当前项和前一项都是 迟到/早退 → 连续违规
        const prevIsVio = isViolation(recordArr[i - 1]);
        const currIsVio = isViolation(recordArr[i]);
        if (prevIsVio && currIsVio) {
            return false;
        }
    }

    // ---------------------- 条件3：任意连续7次，不良考勤≤3次 ----------------------
    // 总条数不足7条，条件自动满足
    if (len < 7) {
        return true;
    }
    // 滑动窗口遍历：窗口大小固定为7，遍历所有可能的连续7条记录
    for (let i = 0; i <= len - 7; i++) {
        // 截取当前窗口的7条记录
        const window7 = recordArr.slice(i, i + 7);
        // 统计窗口内的不良考勤数量
        const badCount = window7.filter(item => isBadRecord(item)).length;
        if (badCount > 3) {
            return false;
        }
    }

    // 所有条件都满足
    return true;
}

// 主逻辑：处理输入 + 批量校验 + 输出结果
let inputLines = [];
readline.on('line', (line) => {
    inputLines.push(line.trim()); // 收集所有输入行
}).on('close', () => {
    const resList = []; // 存储每条考勤的结果(true/false)
    const total = parseInt(inputLines[0]); // 第一行：考勤记录的总条数
    // 遍历每一条考勤记录
    for (let i = 1; i <= total; i++) {
        const recordStr = inputLines[i];
        const recordArr = recordStr.split(' '); // 分割成考勤数组
        const isOk = checkAward(recordArr); // 校验是否达标
        resList.push(isOk ? 'true' : 'false'); // 存入结果数组
    }
    // 按要求输出：结果用空格分隔
    console.log(resList.join(' '));
});
```
