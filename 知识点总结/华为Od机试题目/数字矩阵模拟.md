
题目描述
疫情期间，小明隔离在家，百无聊赖，在纸上写数字玩。他发明了一种写法：
给出数字个数n和行数m（0 < n ≤ 999，0 < m ≤ 999），从左上角的1开始，按照顺时针螺旋向内写方式，依次写出2,3…n，最终形成一个m行矩阵。
小明对这个矩阵有些要求：

每行数字的个数一样多
列的数量尽可能少
填充数字时优先填充外部
数字不够时，使用单个*号占位
输入描述
输入一行，两个整数，空格隔开，依次表示n、m

输出描述
符合要求的唯一矩阵

用例1
输入：

9 4
1
输出：

1 2 3

* * 4
9 * 5
8 7 6

```javascript
// 引入node输入模块，处理控制台输入
const readline = require('readline').createInterface({
    input: process.stdin,
    output: process.stdout
});

// 主逻辑函数
function spiralMatrix(n, m) {
    // 1. 计算矩阵的最小列数c 核心公式 向上取整 得到最小能容纳的列数
    const col = Math.ceil(n / m);
    // 2. 初始化m行col列的矩阵，所有位置默认填充*号
    const matrix = Array(m).fill(0).map(() => Array(col).fill('*'));

    // 3. 定义螺旋填充的边界和变量
    let top = 0;        // 上边界
    let bottom = m - 1; // 下边界
    let left = 0;       // 左边界
    let right = col - 1;// 右边界
    let num = 1;        // 当前要填充的数字

    // 4. 顺时针螺旋填充数字，直到num超过n停止
    while (num <= n) {
        // 方向1：从左到右 → 填充上边界整行
        for (let i = left; i <= right && num <= n; i++) {
            matrix[top][i] = num++;
        }
        top++; // 上边界下移，收缩

        // 方向2：从上到下 → 填充右边界整列
        for (let i = top; i <= bottom && num <= n; i++) {
            matrix[i][right] = num++;
        }
        right--; // 右边界左移，收缩

        // 方向3：从右到左 → 填充下边界整行
        for (let i = right; i >= left && num <= n; i--) {
            matrix[bottom][i] = num++;
        }
        bottom--; // 下边界上移，收缩

        // 方向4：从下到上 → 填充左边界整列
        for (let i = bottom; i >= top && num <= n; i--) {
            matrix[i][left] = num++;
        }
        left++; // 左边界右移，收缩
    }

    // 5. 格式化输出：每行元素用空格拼接，逐行打印
    matrix.forEach(row => {
        console.log(row.join(' '));
    });
}

// 处理输入，调用主函数
readline.on('line', (input) => {
    const [n, m] = input.trim().split(' ').map(Number);
    spiralMatrix(n, m);
    readline.close();
});

```
